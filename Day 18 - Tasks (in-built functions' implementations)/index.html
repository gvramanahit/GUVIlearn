      //actual implementation(to chunk the array) goes here.

          arrayWrap[Math.floor(i/chunk)] = [arrayOne[i]];

Find function(to find the first element which is greater than 5)

let arrayOne =[0,1,2,8,3,7,5];

    let searchVal;
    for(let i=0;i'<'arrayOne.length;i++){
       //actual implementation(to find the first element which is greater than 5) goes here.
       if(arrayOne[i] > 5){
          searchVal = arrayOne[i];
          i = arrayOne.length;
       }
    }
    return searchVal;
}

let foundVal = findArray(arrayOne);
